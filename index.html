async function connectWallet() { try { let ethereum = window.ethereum; // check for Rabby if (window.ethereum && window.ethereum.isRabby) { ethereum = window.ethereum; } if (!ethereum) { alert("No wallet found. Install MetaMask or Rabby."); return; } const accounts = await ethereum.request({ method: "eth_requestAccounts" }); provider = new ethers.providers.Web3Provider(ethereum); signer = provider.getSigner(); userAddress = accounts[0]; document.getElementById("wallet").innerText = "Connected: " + userAddress; } catch (err) { console.log(err); if (err.code === 4001) { alert("You rejected the connection"); } else { alert("Wallet connection failed"); } } }
